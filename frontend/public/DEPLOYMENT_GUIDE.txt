# IntentBrain.ai - Complete VPS Deployment Guide
# =============================================
# Save this file and follow step by step

## PREREQUISITES
- Namecheap VPS with Ubuntu 22.04
- SSH access to your server  
- Domain: intentbrain.ai
- Your VPS IP address

---

## STEP 1: SSH INTO YOUR SERVER

ssh root@YOUR_VPS_IP

---

## STEP 2: RUN THIS ENTIRE BLOCK (Copy & Paste)

# Update system and install everything
sudo apt update && sudo apt upgrade -y && \
sudo apt install -y curl git build-essential nginx certbot python3-certbot-nginx && \
curl -fsSL https://deb.nodesource.com/setup_20.x | sudo -E bash - && \
sudo apt install -y nodejs python3.11 python3.11-venv python3-pip && \
sudo npm install -g pm2

---

## STEP 3: INSTALL MONGODB

curl -fsSL https://www.mongodb.org/static/pgp/server-7.0.asc | sudo gpg -o /usr/share/keyrings/mongodb-server-7.0.gpg --dearmor && \
echo "deb [ arch=amd64,arm64 signed-by=/usr/share/keyrings/mongodb-server-7.0.gpg ] https://repo.mongodb.org/apt/ubuntu jammy/mongodb-org/7.0 multiverse" | sudo tee /etc/apt/sources.list.d/mongodb-org-7.0.list && \
sudo apt update && \
sudo apt install -y mongodb-org && \
sudo systemctl start mongod && \
sudo systemctl enable mongod

---

## STEP 4: CLONE YOUR CODE

sudo mkdir -p /var/www/intentbrain && \
cd /var/www/intentbrain && \
git clone https://github.com/YOUR_USERNAME/YOUR_REPO.git .

** IMPORTANT: Replace YOUR_USERNAME/YOUR_REPO with your actual GitHub repo **

---

## STEP 5: SETUP BACKEND

cd /var/www/intentbrain/backend && \
python3.11 -m venv venv && \
source venv/bin/activate && \
pip install --upgrade pip && \
pip install -r requirements.txt && \
pip install emergentintegrations --extra-index-url https://d33sy5i8bnduwe.cloudfront.net/simple/

---

## STEP 6: CREATE BACKEND .env FILE

cat > /var/www/intentbrain/backend/.env << 'ENVFILE'
MONGO_URL=mongodb://localhost:27017
DB_NAME=intentbrain_prod
EMERGENT_API_KEY=your_emergent_key_here
TWILIO_ACCOUNT_SID=your_twilio_sid
TWILIO_AUTH_TOKEN=your_twilio_token
TWILIO_PHONE_NUMBER=+1234567890
ELEVENLABS_API_KEY=your_elevenlabs_key
STRIPE_SECRET_KEY=your_stripe_live_key
STRIPE_WEBHOOK_SECRET=your_stripe_webhook_secret
JWT_SECRET=change_this_to_random_64_characters
ENVFILE

** IMPORTANT: Edit this file with your real API keys **
nano /var/www/intentbrain/backend/.env

---

## STEP 7: SETUP FRONTEND

cd /var/www/intentbrain/frontend && \
echo "REACT_APP_BACKEND_URL=https://intentbrain.ai" > .env && \
npm install --legacy-peer-deps && \
npm run build

---

## STEP 8: CONFIGURE NGINX

sudo tee /etc/nginx/sites-available/intentbrain << 'NGINXCONF'
server {
    listen 80;
    server_name intentbrain.ai www.intentbrain.ai;

    location / {
        root /var/www/intentbrain/frontend/build;
        index index.html;
        try_files $uri $uri/ /index.html;
    }

    location /api/ {
        proxy_pass http://127.0.0.1:8001;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_read_timeout 86400;
        proxy_send_timeout 86400;
    }

    location /ws/ {
        proxy_pass http://127.0.0.1:8001;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_set_header Host $host;
        proxy_read_timeout 86400;
    }
}
NGINXCONF

sudo ln -sf /etc/nginx/sites-available/intentbrain /etc/nginx/sites-enabled/ && \
sudo rm -f /etc/nginx/sites-enabled/default && \
sudo nginx -t && \
sudo systemctl restart nginx

---

## STEP 9: POINT DOMAIN TO SERVER

In Namecheap DNS Settings:
1. Go to: Domain List > intentbrain.ai > Manage > Advanced DNS
2. Add these A records:

   Type: A    Host: @      Value: YOUR_VPS_IP    TTL: Automatic
   Type: A    Host: www    Value: YOUR_VPS_IP    TTL: Automatic

Wait 5-30 minutes for DNS to propagate.

---

## STEP 10: SETUP SSL (HTTPS)

sudo certbot --nginx -d intentbrain.ai -d www.intentbrain.ai

Follow prompts: enter email, agree to terms, select "redirect HTTP to HTTPS"

---

## STEP 11: START BACKEND WITH PM2

cd /var/www/intentbrain/backend && \
source venv/bin/activate && \
cat > ecosystem.config.js << 'PM2CONF'
module.exports = {
  apps: [{
    name: 'intentbrain-backend',
    script: 'venv/bin/uvicorn',
    args: 'server:app --host 0.0.0.0 --port 8001',
    cwd: '/var/www/intentbrain/backend',
    interpreter: 'none',
    env: {
      PATH: '/var/www/intentbrain/backend/venv/bin:' + process.env.PATH
    }
  }]
};
PM2CONF

pm2 start ecosystem.config.js && \
pm2 save && \
pm2 startup

** Run the command that pm2 startup outputs! **

---

## STEP 12: CONFIGURE STRIPE WEBHOOK

1. Go to: https://dashboard.stripe.com/webhooks
2. Add endpoint: https://intentbrain.ai/api/stripe/webhook
3. Select events: checkout.session.completed, customer.subscription.*
4. Copy webhook signing secret
5. Update STRIPE_WEBHOOK_SECRET in /var/www/intentbrain/backend/.env
6. Run: pm2 restart intentbrain-backend

---

## STEP 13: CONFIGURE TWILIO WEBHOOKS

1. Go to: https://console.twilio.com > Phone Numbers > Your Number
2. Voice webhook: https://intentbrain.ai/api/twilio/voice
3. Status callback: https://intentbrain.ai/api/twilio/status

---

## VERIFY DEPLOYMENT

pm2 status
curl https://intentbrain.ai/api/

Visit https://intentbrain.ai in your browser!
